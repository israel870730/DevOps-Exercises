https://docs.aws.amazon.com/eks/latest/userguide/s3-csi.html#s3-install-driver
https://github.com/awslabs/mountpoint-s3-csi-driver/blob/main/examples/kubernetes/static_provisioning/static_provisioning.yaml

1- Enable aws-mountpoint-s3-csi-driver
aws eks create-addon \
    --cluster-name demo-externaldns \
    --addon-name aws-mountpoint-s3-csi-driver --region us-west-1

2- Check 
aws eks describe-addon \
    --cluster-name demo-externaldns \
    --addon-name aws-mountpoint-s3-csi-driver --region us-west-1

3- Add IAM role al CSI Driver

eksctl create iamserviceaccount \
    --name mountpoint-s3-csi-controller \
    --namespace kube-system \
    --cluster demo-externaldns \
    --role-name MountpointS3CSIRole \
    --attach-policy-arn arn:aws:iam::aws:policy/AmazonS3FullAccess \
    --approve --region us-west-1
